    Data Definition Directives

    Data Definition Directives in assembly language are compiler directives that allocate and initialize memory for variables in a program. These directives define variables with specific sizes and values in the .data or .bss section of an assembly program. They help reserve space for storing numbers, characters, floating-point values, and pointers in memory.

    Data Reservation Directives in .data

    Directive Meaning Size (Bytes) Example Usage
    DB (Define Byte) Reserves 1 byte 1 byte var1 DB 0x12 (1 byte)
    DW (Define Word) Reserves 1 word 2 bytes var2 DW 0x1234 (2 bytes)
    DD (Define Doubleword) Reserves 1 doubleword 4 bytes var3 DD 0x12345678 (4 bytes)
    DQ (Define Quadword) Reserves 1 quadword 8 bytes var4 DQ 0x123456789ABCDEF0 (8 bytes)
    DT (Define Ten Bytes) Reserves 1 ten-byte value 10 bytes var5 DT 123.456 (used for floating-point values in some cases)

    Data Reservation Directives in .bss

    Directive Meaning Size (Bytes) Example Usage
    RESB (Reserve Byte) Reserves N bytes 1 byte buffer RESB 100 (Reserves 100 bytes)
    RESW (Reserve Word) Reserves N words 2 bytes array RESW 10 (Reserves 10 × 2 bytes = 20 bytes)
    RESD (Reserve Doubleword) Reserves N doublewords 4 bytes arr RESD 5 (Reserves 5 × 4 bytes = 20 bytes)
    RESQ (Reserve Quadword) Reserves N quadwords 8 bytes data RESQ 3 (Reserves 3 × 8 bytes = 24 bytes)
    REST (Reserve Ten Bytes) Reserves N ten-byte values 10 bytes bigData REST 2 (Reserves 20 bytes)

    Arithmetic instructions

    add rax, 1    ; Adds 1 to RAX.
sub rax, rdi  ; Subtracts the value of RDI from RAX and stores the result in RAX.
mul rsi       ; Multiplies RAX by RSI and stores the result in RAX.
div rbx       ; Divides RAX by RBX, storing the quotient in RAX and the remainder in RDX.

and rax, rcx  ; Performs a bitwise AND between RAX and RCX, storing the result in RAX.
or rdx, 1     ; Performs a bitwise OR between RDX and 1, storing the result in RDX.
xor rsi, rsi  ; Performs a bitwise XOR between RSI and itself (zeroing RSI).
not rbx       ; Inverts every bit of RBX.

    System calls

    Read

    • rax Syscall number (0 for read)
    • rdi File descriptor (0 for stdin)
    • rsi Buffer to store the input
    • rdx Number of bytes to read

    Return : Number of bytes read in rax, 0 if EOF

    section .bss
    buffer resb 10  ; Reserve 10 bytes for input
...
    mov rax, 0       ; sys_read
    mov rdi, 0       ; stdin
    mov rsi, buffer  ; buffer address
    mov rdx, 10      ; read up to 10 bytes
    syscall

    Write

    • rax Syscall number (1 for write)
    • rdi File descriptor (1 for standard output)
    • rsi Pointer to the string (buffer) to write
    • rdx Number of bytes to write

    Return : Number of bytes written in rax

    Example: Writing message to stdout

        mov rax, 1         ; sys_write
    mov rdi, 1         ; stdout
    mov rsi, msg       ; message address
    mov rdx, msg_len   ; message length
    syscall

    Open

    • rax Syscall number (2 for open)
    • rdi Pointer to filename
    • rsi Flags (0 = read-only, 1 = write-only, 2 = read/write)
    • rdx File permissions (only if creating a file)

    Return : File descriptor in rax

    Example: Open a file for reading

        mov rax, 2        ; sys_open
    mov rdi, filename ; file name
    mov rsi, 0        ; O_RDONLY
    syscall

    Example: Creating or append a file

        mov rax, 2  ; sys_open
    mov rdi, filename  ; file name
    mov rsi, 0x441 ; O_WRONLY | O_CREAT | O_APPEND
    mov rdx, 0o666 ; rw-rw-rw-
    syscall

    Close

    • rax Syscall number (3 for close)
    • rdi File descriptor to close

    Example: Closing a file descriptor (fd)

        mov rax, 3  ; sys_close
    mov rdi, fd ; file descriptor
    syscall

    Exit

    • rax Syscall number (60 for exit)
    • rdi Exit status (0 = success, nonzero = error)

    Example: Exit with status 0

        mov rax, 60  ; sys_exit
    xor rdi, rdi ; exit(0)
    syscall

    Socket

    • rax Syscall number (41 for socket)
    • rdi Address family (2 for AF_INET)
    • rsi Socket type (1 for SOCK_STREAM)
    • rdx Protocol (0 for default)

    Return : Socket file descriptor in rax

    Example: Create an IPv4 TCP socket

        mov rax, 41     ; sys_socket
    mov rdi, 2      ; AF_INET
    mov rsi, 1      ; SOCK_STREAM
    xor rdx, rdx    ; protocol 0
    syscall

    Bind

    • rax Syscall number (49 for bind)
    • rdi Socket file descriptor
    • rsi Pointer to sockaddr structure
    • rdx Size of the sockaddr structure

    Example: Bind a socket

    section .data
    sockaddr_in:
        .sin_family dw 2 ; AF_INET
        .sin_port   dw 0x3905 ; Port 1337 (little-endian: 0x0539)
        .sin_addr   dd 0 ; INADDR_ANY (0.0.0.0)
...
    mov rax, 49      ; sys_bind
    mov rdi, sockfd  ; socket file descriptor
    mov rsi, sockaddr; pointer to sockaddr structure
    mov rdx, 16      ; size of sockaddr_in
    syscall

    Recvfrom

    • rax Syscall number (45 for recvfrom)
    • rdi Socket file descriptor
    • rsi Pointer to buffer for received data
    • rdx Maximum number of bytes to receive
    • r10 Flags
    • r8 Pointer to sender’s address
    • r9 Pointer to sender’s address length

    Return : Number of bytes received in rax

    Example: Receive data from a socket

        mov rax, 45      ; sys_recvfrom
    mov rdi, sockfd  ; socket descriptor
    mov rsi, buffer  ; buffer address
    mov rdx, 1024    ; max bytes to receive
    xor r10, r10     ; flags = 0
    xor r8, r8       ; no sender address
    xor r9, r9       ; no sender length
    syscall
